例子是一个分页组件，封装element ui分页组件的传值，使用时直接调用就行（封装意义在于多处用到

<template>
  <el-pagination
    @size-change="handleSizeChange"      //点击显示数目赋值
    @current-change="handlePageChange"  //点击页数赋值
    :current-page="config.page"  // 当前是第几页
    :page-size="config.size"     // 用户选择一页显示多少条数据
    :page-sizes="5"   // 应为一个数组存放一页显示多少条数据，初始化默认选择第一项
   :total="total"              //总条目数
    layout="total, sizes, prev, pager, next, jumper"
  ></el-pagination>
</template>

<script>
export default {
  name: 'share-pagination',
  //用props接收父组件传递的参数，父组件里为 :total='total'动态绑定数据，传递给子组件
  props: {
    total: {
      type: Number,
      default: 0
    }
  },
  data () {
    return {
      config: {}  // 存放page和size
    }
  },
  mounted () {
    this.initPage()  //默认加载初始化方法
  },
  methods: {
    setConfig (config) { //接收到父组件传来的值
      for (const key in config) {
        this.config[key] = config[key] //更新组件本地值
      }
      this.$emit('change', this.config) //更新
    },
    initPage () {
      this.config = {  // total接收父组件传的值，page和size自动初始化值
        total: this.total,
        page: 1,
        size: this.$utils.getPageSizeOptions()[0]
      }
      this.$emit('change', this.config, true)
    },
    handleSizeChange (val) {  //改变页数/分页时，改变本地值
      this.config.size = val
      this.configChange()
    },
    handlePageChange (val) {   //改变页数/分页时，改变本地值
      this.config.page = val
      this.configChange()
    },
    configChange () {
      this.config.total = this.total
      this.$emit('change', this.config)  
    // 用$emit子到父组件通信，第一个参数change为父组件里调用子组件定义的接收函数的事件
    // 例如<子组件 :change="自定义函数"></子组件>第二个参数为传的值
    }
  }
}
</script>

贴个精简的父子组件通信例子吧

  • 父组件向子组件传值

    父组件定义:lists='list'，:alert1='alert1'
子组件用props:{'lists:{}]','alert1:[]}'}接收

    注意props传值有两种情况，一种是固定传值，第二种是动态传值

    关于第二种动态传值，有两种方法

    1. 用v-if 定义一个flag，初始为false，在动态改变了username的时候flag设置为true

      // 父组件
<children :username="username" v:if="flag"/>
// 子组件
props: ['username']

    2. 子组件通过watch监听数据变化

            <input type="text" v-model="childrenData">
<script>
    export default {
        name: "children",
      props:{
          fatherData:String // 父组件绑定传的值
      },
      data(){
          return{
            childrenData:''
          }
      },
      created(){
        this.childrenData=this.fatherData
      },

      //这里用watch方法来监听父组件传过来的值，来实现实时更新
      watch:{
        //fatherData为props拿到的父组件值
        fatherData(val){
          this.childrenData = val    
        //将父组件的值赋给childrenData 子组件的值
        }
      }
    }

  • 子组件中不使用props来接收父组件的方法

    子组件在methods中通过定义一个新方法，在方法体里面写this.$emit(‘父组件事件’, ‘参数1’)

    以这种形式获取父组件的方法，在赋值给子组件新定义的方法

    注意alert1是父组件中绑定时候自己定义的名字

    canshu1若该方法带参数则加上，不带则加空字符串即可

    //在父组件里
<aler :data='data' @alert1='sendData'></aler>

    return "这是父组件的值"                
}
//在子组件里
methods: {
    alert1 () {
        let data = this.$emit('alert1','')
    }
}

  • 简单概括

    <msg ref="msg" @myChange='change></msg> //子组件触发change方法
    methods: {
        //父组件通过ref调用子组件方法
        ale() {  
          this.$refs.msg.alert1() // 调用成功
        }
        change(val) {   
            console.log(val) // [1,2,3]
        }
    }
    //子组件里方法
    alert1() {
        alert('调用成功')
    }
    change() {
        this.$emit('myChange',[1,2,3]) //监听自定义myChange事件
    }
